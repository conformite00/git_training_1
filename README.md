README파일입니다
시작하자마자 git init
.gitignore 파일, README.md 파일 생성
git add.
git commit -m 'first' //첫번째 커밋 작성
README 파일 내용 채운 후
git commit -m 'senond' //두번째 커밋 작성
no.txt 파일 생성 후
.gitignore 파일에 no.txt 추가 후
git commit -m 'third' //세번째 커밋 작성
git remote add origin (주소) //연결하기
git push origin master //업로드하기
